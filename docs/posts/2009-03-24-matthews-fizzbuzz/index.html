#!/bin/bash

before=”$(date -j -f “%a %b %d %T %Z %Y” “`date`” “+%s”)”

echo “=========================Starting Fizzbuzz: $before=========================”

num=1

while [ $num -lt 101 ]; do

#if % 3&&5 print fizzbuzz

if [ $(( $num % 5 )) -eq 0 ] && [ $(( $num % 3 )) -eq 0 ] ; then

echo “fizzbuzz”

elif [ $(( $num % 5 )) -eq 0 ] ; then

echo “buzz”

elif [ $(( $num %3 )) -eq 0 ] ; then

echo fizz

else

echo $num

fi

let num=num+1

done

after=”$(date -j -f “%a %b %d %T %Z %Y” “`date`” “+%s”)”

elapsed=”$(expr $after – $before)”

echo “Time elapsed: $elapsed”

echo “=========================End Fizzbuzz: $after=========================”

exit 0
